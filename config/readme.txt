CREATE VIEW LAPORAN_UPLOAD_HARIAN (ID_UPLOAD, JUMLAH) AS
SELECT TGL_UPLOAD, COUNT(*) AS "JUMLAH"
FROM TB_FILE
GROUP BY TGL_UPLOAD
ORDER BY TGL_UPLOAD ASC;

---------------------------------------------
CREATE VIEW LAPORAN_UPLOAD_MINGGUAN (TAHUN_MINGGU, JUMLAH) AS
SELECT "TAHUN_MINGGU", COUNT(*) AS "JUMLAH"
FROM (
SELECT TO_CHAR(TB_FILE.TGL_UPLOAD, 'YYYY-IW') AS "TAHUN_MINGGU"
FROM TB_FILE
) DT
GROUP BY "TAHUN_MINGGU"
ORDER BY "TAHUN_MINGGU" ASC;

-----------

CREATE VIEW LAPORAN_UPLOAD_BULANAN (TAHUN_BULAN, JUMLAH) AS
SELECT "TAHUN_BULAN", COUNT(*) AS "JUMLAH"
FROM (
SELECT TO_CHAR(TB_FILE.TGL_UPLOAD, 'YYYY-MM') AS "TAHUN_BULAN"
FROM TB_FILE
) DT
GROUP BY "TAHUN_BULAN"
ORDER BY "TAHUN_BULAN" ASC;

--------------------------

CREATE VIEW LAPORAN_UPLOAD_TAHUNAN (TAHUN, JUMLAH) AS
SELECT "TAHUN", COUNT(*) AS "JUMLAH"
FROM (
SELECT TO_CHAR(TB_FILE.TGL_UPLOAD, 'YYYY') AS "TAHUN"
FROM TB_FILE
) DT
GROUP BY "TAHUN"
ORDER BY "TAHUN" ASC;
